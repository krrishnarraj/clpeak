add_library(peak_common STATIC
    peak.cpp
    common.cpp
    backend_gating.cpp
    calibrate.cpp
    inventory.cpp
    logger.cpp
    options.cpp
    result_store.cpp
)
# logger.cpp lives in src/cli/ (desktop) or android/ (Android JNI) —
# each consumer brings its own output implementation.
target_include_directories(peak_common PUBLIC
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_SOURCE_DIR}/include/common
)
target_compile_features(peak_common PUBLIC cxx_std_11)
if(CMAKE_CXX_COMPILER_ID MATCHES "Clang|GNU|AppleClang")
    target_compile_options(peak_common PRIVATE -fPIC -Wall -Wextra -Wno-unused-parameter)
endif()

# version.h is generated by the version custom target
include(${CMAKE_SOURCE_DIR}/cmake/version.cmake)
clpeak_setup_version(peak_common)
